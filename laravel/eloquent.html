Modul 9: Eloquent ORM

Estimasi: 70 Menit
Teori & Praktikum

Eloquent ORM (Object-Relational Mapper) adalah jembatan antara dunia objek PHP dan tabel database relasional. Dengan Eloquent, Anda tidak menulis SQL — cukup pakai sintaks PHP yang ekspresif. Setiap tabel diwakili oleh sebuah Model, dan setiap baris menjadi instance Model.

🌐
Analogi: Penerjemah Bahasa

Anda bicara bahasa PHP ("ambilkan user yang aktif"). Eloquent menerjemahkan ke bahasa SQL ("SELECT * FROM users WHERE active=1"). Database jawab dengan baris-baris data, Eloquent bungkus jadi objek PHP yang bisa Anda akses dengan $user->name. Anda tidak perlu fasih bahasa SQL — Eloquent yang tangani.

Membuat Model

Pakai Artisan dengan flag -m agar sekaligus dibuat file migration-nya:

Terminal 
# Model + Migration
php artisan make:model Post -m

# Model + Migration + Controller + Resource + Factory + Seeder (paling lengkap)
php artisan make:model Post -a
Konvensi Penamaan

Model bernama singular dengan PascalCase (Post, User, BlogCategory). Eloquent otomatis mencari tabel dengan nama plural snake_case (posts, users, blog_categories). Override dengan protected $table = 'nama_custom'; jika perlu.

Anatomi Model

app/Models/Post.php 
<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\Factories\HasFactory;

class Post extends Model
{
    use HasFactory;

    // Kolom yang BOLEH diisi via Mass Assignment
    protected $fillable = ['title', 'body', 'user_id'];

    // Atau sebaliknya, kolom yang DILINDUNGI (sisanya boleh)
    // protected $guarded = ['id'];

    // Konversi tipe otomatis
    protected $casts = [
        'is_published' => 'boolean',
        'published_at' => 'datetime',
        'meta'         => 'array', // JSON → array PHP
    ];

    // Sembunyikan dari output JSON (mis. password)
    protected $hidden = ['password', 'remember_token'];
}

Operasi CRUD dengan Eloquent


            
📖 Read — Membaca Data

            

                

                    

                    contoh query
                    
                

                
// Ambil semua
$posts = Post::all();

// Find berdasarkan primary key
$post = Post::find(1);             // null jika tidak ada
$post = Post::findOrFail(1);       // 404 jika tidak ada

// Where clause (chainable)
$active = Post::where('status', 'published')
              ->where('views', '>', 100)
              ->orderBy('created_at', 'desc')
              ->take(10)
              ->get();

// Get first / latest / oldest
$latest = Post::latest()->first();   // = orderBy created_at desc + first
$oldest = Post::oldest()->first();

// Aggregate
$count = Post::count();
$max   = Post::max('views');
$avg   = Post::avg('views');
$sum   = Post::sum('views');

// Pagination otomatis
$posts = Post::paginate(15); // 15 per halaman

// Ambil hanya 1 kolom
$titles = Post::pluck('title'); // Collection of strings

// Where ekstra
Post::whereIn('id', [1, 2, 3])->get();
Post::whereBetween('views', [100, 500])->get();
Post::whereNull('deleted_at')->get();
Post::whereDate('created_at', today())->get();

            


            
✏️ Create — Membuat Data

            
Sebelum pakai mass assignment, definisikan $fillable di Model untuk keamanan:

            

                

                    

                    contoh create
                    
                

                
// Cara 1: Instansiasi + save
$post = new Post;
$post->title = 'Belajar Eloquent';
$post->body  = 'Eloquent itu menyenangkan.';
$post->save();

// Cara 2: Mass assignment dengan create() — paling singkat
$post = Post::create([
    'title' => 'Belajar Eloquent',
    'body'  => 'Eloquent itu menyenangkan.',
]);

// Cara 3: firstOrCreate — buat jika belum ada
$cat = Category::firstOrCreate(
    ['name' => 'Tech'],         // cari berdasarkan ini
    ['description' => '...']    // kolom tambahan saat create
);

// Cara 4: updateOrCreate — update jika ada, create jika tidak
Setting::updateOrCreate(
    ['key' => 'theme'],
    ['value' => 'dark']
);

            


            
🔄 Update — Mengubah Data

            

                

                    

                    contoh update
                    
                

                
// Update satu record
$post = Post::find(1);
$post->title = 'Judul Baru';
$post->save();

// Atau singkat:
Post::find(1)->update(['title' => 'Judul Baru']);

// Update massal (banyak record sekaligus)
Post::where('status', 'draft')
    ->update(['status' => 'archived']);

// Increment/decrement (counter, view, like)
Post::find(1)->increment('views');          // +1
Post::find(1)->increment('views', 10);      // +10
Product::find(1)->decrement('stock');     // -1

            


            
🗑 Delete — Menghapus Data

            

                

                    

                    contoh delete
                    
                

                
// Delete satu instance
$post = Post::find(1);
$post->delete();

// Delete langsung by ID
Post::destroy(1);
Post::destroy([1, 2, 3]);

// Delete massal dengan kondisi
Post::where('status', 'draft')->delete();

// Soft delete (pakai trait SoftDeletes di model)
$post->delete();                // set deleted_at, tidak benar2 hapus
$post->restore();               // kembalikan
$post->forceDelete();           // hapus permanen
Post::withTrashed()->get();    // termasuk yang soft-deleted

            


            
Accessor & Mutator (Property Cerdas)

            
Accessor = modifikasi data saat dibaca. Mutator = modifikasi saat ditulis. Berguna untuk formatting, hashing, dll.

            

                

                    

                    app/Models/User.php
                

                
use Illuminate\Database\Eloquent\Casts\Attribute;

class User extends Model
{
    // Accessor: $user->full_name → "Arif Hidayah"
    protected function fullName(): Attribute
    {
        return Attribute::make(
            get: fn () => $this->first_name . ' ' . $this->last_name
        );
    }

    // Mutator: hash password otomatis saat di-set
    protected function password(): Attribute
    {
        return Attribute::make(
            set: fn ($value) => bcrypt($value)
        );
    }
}

// Pemakaian:
echo $user->full_name;        // → "Arif Hidayah"
$user->password = 'rahasia';   // → otomatis di-hash sebelum simpan

            


            
Scope (Query yang Reusable)

            
Daripada menulis where('status', 'published') berulang, buat scope di Model:

            

                

                    

                    app/Models/Post.php
                

                
class Post extends Model
{
    // Scope: published
    public function scopePublished($query)
    {
        return $query->where('status', 'published');
    }

    // Scope dengan parameter
    public function scopePopular($query, $min = 100)
    {
        return $query->where('views', '>=', $min);
    }
}

// Pemakaian — drop "scope" dan camelCase jadi lowercase:
$posts = Post::published()->popular(500)->get();
// Setara dengan: where('status', 'published')->where('views', '>=', 500)
